/** Αφαιρεί διπλότυπα από τον πίνακα hasGenre**/
with tempTable as(
  Select *,ROW_NUMBER()  OVER (partition by movie_id,genre_id ORDER BY movie_id) counter
  from hasGenre)
delete from tempTable where counter>1;
